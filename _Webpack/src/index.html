Create a new package.json

npm init -y

Install Webpack (local)

npm install -D webpack webpack-cli

-D добавит в зависимости devDependencies: webpack и webpack-cli

Create a new webpack.config.js

webpack.config.js

Add minimal config

const path = require('path')

module.exports = {

entry: './src/index.js',

output: {

filename: 'bundle.js',

path: path.resolve(__dirname, 'dist')

}

};

First run

npx webpack --config webpack.config.js

Create a plugin

npm i -D html-webpack-plugin

Create a plugin for clear hash

npm i -D clean-webpack-plugin

Create style plugins

npm i -D style-loader css-loader

Create style normalize

npm i normalize.css

Create file loader

npm i -D file-loader

Create xml loader

npm i -D xml-loader

Create csv loader

npm i -D csv-loader
