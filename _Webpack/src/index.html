Create a new package.json

npm init -y

Install Webpack (local)

npm install -D webpack webpack-cli

-D добавит в зависимости devDependencies: webpack и webpack-cli

Create a new webpack.config.js

webpack.config.js

Add minimal config

const path = require('path')

module.exports = {

entry: './src/index.js',

output: {

filename: 'bundle.js',

path: path.resolve(__dirname, 'dist')

}

};

First run

npx webpack --config webpack.config.js

Install a plugin

npm i -D html-webpack-plugin

Install a plugin for clear hash

npm i -D clean-webpack-plugin

Install style plugins

npm i -D style-loader css-loader

Install style normalize

npm i normalize.css

Install file loader

npm i -D file-loader

Install xml loader

npm i -D xml-loader

Install csv loader

npm i -D csv-loader

Install jquery

npm i -S jquery

Install dev server

npm i -D webpack-dev-server

"start": "webpack serve --mode development --open"

Install copy-webpack-plugin

npm i -D copy-webpack-plugin

Install mini-css-extract-plugin

npm i -D mini-css-extract-plugin

Install cross-env

npm i -D cross-env

Install terser-webpack-plugin

npm i -D terser-webpack-plugin

Install optimize-css-assets-webpack-plugin

npm i -D optimize-css-assets-webpack-plugin

Install less

npm i -D less-loader

npm i -D less

Install sass

npm i -D sass-loader sass

Install babel

npm i -D babel-loader @babel/core

npm i -D @babel/preset-env

npm i -save @babel/polyfill

npm i -D @babel/preset-typescript

npm i -D @babel/preset-react

npm i react react-dom
